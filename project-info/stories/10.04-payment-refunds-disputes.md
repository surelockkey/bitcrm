## 1. System Overview

### 1.1. Purpose

Implement comprehensive refund and dispute management functionality within Bitrix24 to handle:

- Full and partial refunds through Stripe API
- Automated chargeback dispute tracking and management
- Evidence submission for dispute resolution
- Timeline integration for all refund/dispute activities
- Manager notifications for critical dispute events

**Key Features:**
- One-click refund processing from Deal card
- Automated Dispute Pipeline creation
- Evidence upload to Stripe
- Real-time webhook synchronization
- Complete audit trail in CRM Timeline

### 1.2. Business Requirements

**Why Refund & Dispute Management?**

**Refunds:**
- Customer service recovery (mistakes, double charges)
- Service quality issues
- Contract cancellations
- Goodwill gestures

**Disputes (Chargebacks):**
- Customer disputes charge with bank
- Company has 7-10 days to respond with evidence
- Losing dispute = lost revenue + $15 fee
- Winning dispute = revenue retained + fee refunded

## 2. Refund Management

### 2.1. Refund Types

**Full Refund:**
- Entire payment amount returned to customer
- Use cases: Order cancellation, service not provided, duplicate charge

**Partial Refund:**
- Specific amount returned (less than full payment)
- Use cases: Service quality adjustment, partial service completion, promotional credit

**Refund Limitations:**
- Can only refund successful payments
- Refund must be ≤ original charge amount
- Stripe fees (2.9% + $0.30) are NOT refunded by Stripe
- ACH refunds can take 5-10 business days to appear

### 2.3. Technical Implementation

**File:** `/local/modules/payment/lib/refund.php`

```php
<?php
namespace Local\Payment;

use Bitrix\Main\Type\DateTime;

class RefundManager {

    /**
     * Process refund for a payment
     *
     * @param int $dealId Deal ID
     * @param string $paymentIntentId Stripe Payment Intent ID
     * @param float $amount Refund amount (null = full refund)
     * @param string $reason Refund reason
     * @param string $note Internal note
     * @return array ['success' => bool, 'refund_id' => string, 'error' => string]
     */
    public static function processRefund($dealId, $paymentIntentId, $amount = null, $reason = '', $note = '') {
        global $DB;

        // Initialize Stripe
        \Stripe\Stripe::setApiKey(\COption::GetOptionString('stripe', 'secret_key'));

        try {
            // Get original payment details
            $paymentIntent = \Stripe\PaymentIntent::retrieve($paymentIntentId);
            $originalAmount = $paymentIntent->amount / 100;

            // Validate refund amount
            if ($amount !== null && $amount > $originalAmount) {
                throw new \Exception('Refund amount cannot exceed original charge');
            }

            // Create refund
            $refundParams = [
                'payment_intent' => $paymentIntentId,
                'metadata' => [
                    'bitrix_deal_id' => $dealId,
                    'refund_reason' => $reason,
                    'refund_note' => $note,
                    'processed_by' => $GLOBALS['USER']->GetID()
                ]
            ];

            // Add amount for partial refund
            if ($amount !== null) {
                $refundParams['amount'] = intval($amount * 100); // Convert to cents
            }

            // Map reason to Stripe reason codes
            $refundParams['reason'] = self::mapReasonToStripe($reason);

            // Create refund in Stripe
            $refund = \Stripe\Refund::create($refundParams);

            // Log refund in custom table
            $DB->Query("
                INSERT INTO b_payment_refunds
                (DEAL_ID, PAYMENT_INTENT_ID, REFUND_ID, AMOUNT, REASON, NOTE, STATUS, CREATED_BY, CREATED_DATE)
                VALUES (
                    {$dealId},
                    '{$DB->ForSql($paymentIntentId)}',
                    '{$DB->ForSql($refund->id)}',
                    " . ($amount ?? $originalAmount) . ",
                    '{$DB->ForSql($reason)}',
                    '{$DB->ForSql($note)}',
                    'processing',
                    " . $GLOBALS['USER']->GetID() . ",
                    " . $DB->CurrentTimeFunction() . "
                )
            ");

            // Update Deal payment status immediately
            $newStatus = ($amount !== null && $amount < $originalAmount)
                ? 'Partially Refunded'
                : 'Refunded';

            \CCrmDeal::Update($dealId, [
                'UF_CRM_PAYMENT_STATUS' => $newStatus
            ]);

            // Add timeline entry
            self::addRefundTimelineEntry($dealId, $refund, $reason, $note);

            return [
                'success' => true,
                'refund_id' => $refund->id,
                'error' => null
            ];

        } catch (\Exception $e) {
            return [
                'success' => false,
                'refund_id' => null,
                'error' => $e->getMessage()
            ];
        }
    }

    /**
     * Map internal reason to Stripe reason code
     */
    private static function mapReasonToStripe($reason) {
        $reasonMap = [
            'Duplicate charge' => 'duplicate',
            'Requested by customer' => 'requested_by_customer',
            'Service not provided' => 'fraudulent', // Disputed as not received
            'Service quality issue' => 'requested_by_customer',
            'Other' => 'requested_by_customer'
        ];

        return $reasonMap[$reason] ?? 'requested_by_customer';
    }

    /**
     * Add refund entry to Deal timeline
     */
    private static function addRefundTimelineEntry($dealId, $refund, $reason, $note) {
        $amount = $refund->amount / 100;
        $refundType = ($refund->amount < $refund->charge->amount) ? 'Partial' : 'Full';

        $comment = "💸 {$refundType} refund processed: $" . number_format($amount, 2) . "\n";
        $comment .= "Refund ID: {$refund->id}\n";
        $comment .= "Reason: {$reason}\n";

        if (!empty($note)) {
            $comment .= "Note: {$note}\n";
        }

        $comment .= "Processed by: " . $GLOBALS['USER']->GetFullName();

        \Bitrix\Crm\Timeline\CommentController::getInstance()->onCreate([
            'ENTITY_TYPE_ID' => \CCrmOwnerType::Deal,
            'ENTITY_ID' => $dealId,
            'COMMENT' => $comment
        ]);
    }

    /**
     * Get refund history for Deal
     */
    public static function getRefundHistory($dealId) {
        global $DB;

        $result = $DB->Query("
            SELECT r.*, u.NAME, u.LAST_NAME
            FROM b_payment_refunds r
            LEFT JOIN b_user u ON r.CREATED_BY = u.ID
            WHERE r.DEAL_ID = {$dealId}
            ORDER BY r.CREATED_DATE DESC
        ");

        $refunds = [];
        while ($row = $result->Fetch()) {
            $refunds[] = $row;
        }

        return $refunds;
    }
}
```

```php
<?php
/**
 * Handle refund.succeeded webhook
 */
function handleRefundSucceeded($event) {
    global $DB;

    $refund = $event->data->object;
    $refundId = $refund->id;
    $dealId = $refund->metadata->bitrix_deal_id ?? null;

    if (!$dealId) {
        // Try to get Deal ID from Payment Intent
        $paymentIntent = \Stripe\PaymentIntent::retrieve($refund->payment_intent);
        $dealId = $paymentIntent->metadata->bitrix_deal_id ?? null;
    }

    if (!$dealId) {
        error_log("Refund webhook: No Deal ID found for refund {$refundId}");
        return;
    }

    // Update refund status in database
    $DB->Query("
        UPDATE b_payment_refunds
        SET STATUS = 'succeeded',
            UPDATED_DATE = " . $DB->CurrentTimeFunction() . "
        WHERE REFUND_ID = '{$DB->ForSql($refundId)}'
    ");

    // Add timeline entry
    $amount = $refund->amount / 100;
    \Bitrix\Crm\Timeline\CommentController::getInstance()->onCreate([
        'ENTITY_TYPE_ID' => \CCrmOwnerType::Deal,
        'ENTITY_ID' => $dealId,
        'COMMENT' => "✅ Refund confirmed by Stripe: $" . number_format($amount, 2) . "\n"
                   . "Funds will appear in customer account in 5-10 business days.\n"
                   . "Refund ID: {$refundId}"
    ]);

    // Send email notification to customer
    $deal = \CCrmDeal::GetByID($dealId);
    if ($deal && $deal['CONTACT_ID']) {
        $contact = \CCrmContact::GetByID($deal['CONTACT_ID'], false);
        if ($contact && !empty($contact['EMAIL'])) {
            sendRefundConfirmationEmail($contact['EMAIL'], $amount, $refundId);
        }
    }
}

/**
 * Handle refund.failed webhook
 */
function handleRefundFailed($event) {
    global $DB;

    $refund = $event->data->object;
    $refundId = $refund->id;
    $dealId = $refund->metadata->bitrix_deal_id ?? null;

    if (!$dealId) {
        return;
    }

    // Update refund status
    $DB->Query("
        UPDATE b_payment_refunds
        SET STATUS = 'failed',
            ERROR_MESSAGE = '{$DB->ForSql($refund->failure_reason)}',
            UPDATED_DATE = " . $DB->CurrentTimeFunction() . "
        WHERE REFUND_ID = '{$DB->ForSql($refundId)}'
    ");

    // Add timeline entry
    \Bitrix\Crm\Timeline\CommentController::getInstance()->onCreate([
        'ENTITY_TYPE_ID' => \CCrmOwnerType::Deal,
        'ENTITY_ID' => $dealId,
        'COMMENT' => "❌ Refund FAILED: $" . ($refund->amount / 100) . "\n"
                   . "Reason: " . $refund->failure_reason . "\n"
                   . "Please contact support."
    ]);

    // Notify manager
    $deal = \CCrmDeal::GetByID($dealId);
    $managerId = $deal['ASSIGNED_BY_ID'] ?? 1;

    \CIMNotify::Add([
        'TO_USER_ID' => $managerId,
        'NOTIFY_TYPE' => \IM_NOTIFY_SYSTEM,
        'NOTIFY_MODULE' => 'crm',
        'NOTIFY_MESSAGE' => "⚠️ Refund FAILED for Deal #{$dealId}. "
                          . "Reason: " . $refund->failure_reason
    ]);
}
```

## 3. Dispute Management

### 3.1. Dispute Pipeline

**Pipeline Structure:**

```
CRM → Deals → Category: Disputes

Stages:
1. NEW_DISPUTE        (Red)     - Just received from Stripe
```

```php
// File: /local/php_interface/include/stripe_webhooks.php

/**
 * Handle charge.dispute.created webhook
 */
function handleDisputeCreated($event) {
    $dispute = $event->data->object;
    $originalDealId = $dispute->metadata->bitrix_deal_id ?? null;

    if (!$originalDealId) {
        // Try to get from charge
        $charge = \Stripe\Charge::retrieve($dispute->charge);
        $originalDealId = $charge->metadata->bitrix_deal_id ?? null;
    }

    if (!$originalDealId) {
        error_log("Dispute webhook: No Deal ID found for dispute {$dispute->id}");
        return;
    }

    // Get original Deal info
    $originalDeal = \CCrmDeal::GetByID($originalDealId);
    if (!$originalDeal) {
        return;
    }

    // Create Deal in Dispute Pipeline
    $disputeDeal = new \CCrmDeal();
    $newDisputeId = $disputeDeal->Add([
        'TITLE' => "Dispute: " . $dispute->id . " - " . $originalDeal['TITLE'],
        'STAGE_ID' => 'NEW_DISPUTE',
        'CATEGORY_ID' => DISPUTE_PIPELINE_ID, // Set in config
        'COMPANY_ID' => $originalDeal['COMPANY_ID'],
        'CONTACT_ID' => $originalDeal['CONTACT_ID'],
        'ASSIGNED_BY_ID' => getShiftManagerId(),
        'OPPORTUNITY' => $dispute->amount / 100,
        'CURRENCY_ID' => 'USD',

        // Custom fields
        'UF_CRM_ORIGINAL_DEAL_ID' => $originalDealId,
        'UF_CRM_DISPUTE_ID' => $dispute->id,
        'UF_CRM_DISPUTE_AMOUNT' => $dispute->amount / 100,
        'UF_CRM_DISPUTE_REASON' => $dispute->reason,
        'UF_CRM_DISPUTE_STATUS' => $dispute->status,
        'UF_CRM_EVIDENCE_DUE_DATE' => date('d.m.Y H:i:s', $dispute->evidence_details->due_by),
        'UF_CRM_DISPUTE_FEE' => 15.00
    ]);

    if (!$newDisputeId) {
        error_log("Failed to create Dispute Deal: " . $disputeDeal->LAST_ERROR);
        return;
    }

    // Add timeline entry to original Deal
    \Bitrix\Crm\Timeline\CommentController::getInstance()->onCreate([
        'ENTITY_TYPE_ID' => \CCrmOwnerType::Deal,
        'ENTITY_ID' => $originalDealId,
        'COMMENT' => "⚠️ CHARGEBACK DISPUTE OPENED\n"
                   . "Amount: $" . number_format($dispute->amount / 100, 2) . "\n"
                   . "Reason: " . $dispute->reason . "\n"
                   . "Dispute ID: {$dispute->id}\n"
                   . "[View Dispute Deal](/crm/deal/details/{$newDisputeId}/)"
    ]);

    // Create urgent Task for Manager
    $dueDate = date('d.m.Y H:i:s', $dispute->evidence_details->due_by - 86400); // 1 day before

    $taskId = \CTaskItem::Add([
        'TITLE' => "URGENT: Respond to chargeback for Deal #{$originalDealId}",
        'DESCRIPTION' => "A customer has disputed a charge with their bank.\n\n"
                       . "Amount: $" . number_format($dispute->amount / 100, 2) . "\n"
                       . "Reason: {$dispute->reason}\n"
                       . "Evidence due by: " . date('m/d/Y g:i A', $dispute->evidence_details->due_by) . "\n\n"
                       . "You must respond with evidence before the deadline or we will automatically lose this dispute.\n\n"
                       . "View Dispute: https://bitrix24.com/crm/deal/details/{$newDisputeId}/",
        'RESPONSIBLE_ID' => getShiftManagerId(),
        'DEADLINE' => $dueDate,
        'PRIORITY' => 2, // High
        'UF_CRM_TASK' => ['D_' . $newDisputeId],
        'TAGS' => ['URGENT', 'DISPUTE', 'CHARGEBACK']
    ], $GLOBALS['USER']->GetID());

    // Send notifications to all managers
```

```php
function handleDisputeClosed($event) {
    $dispute = $event->data->object;
    $disputeId = $dispute->id;

    $disputeDeal = findDisputeDealByStripeId($disputeId);
    if (!$disputeDeal) {
        return;
    }

    if ($dispute->status === 'won') {
        // Dispute WON
        \CCrmDeal::Update($disputeDeal['ID'], [
            'STAGE_ID' => 'DISPUTE_WON',
            'UF_CRM_DISPUTE_STATUS' => 'Won'
        ]);

        \Bitrix\Crm\Timeline\CommentController::getInstance()->onCreate([
            'ENTITY_TYPE_ID' => \CCrmOwnerType::Deal,
            'ENTITY_ID' => $disputeDeal['ID'],
            'COMMENT' => "✅ DISPUTE WON!\n"
                       . "Amount retained: $" . number_format($dispute->amount / 100, 2) . "\n"
                       . "$15 dispute fee will be refunded by Stripe"
        ]);

        // Notify original Deal
        $originalDealId = $disputeDeal['UF_CRM_ORIGINAL_DEAL_ID'];
        \Bitrix\Crm\Timeline\CommentController::getInstance()->onCreate([
            'ENTITY_TYPE_ID' => \CCrmOwnerType::Deal,
            'ENTITY_ID' => $originalDealId,
            'COMMENT' => "✅ Chargeback dispute WON - payment retained"
        ]);

    } else {
        // Dispute LOST
        \CCrmDeal::Update($disputeDeal['ID'], [
            'STAGE_ID' => 'DISPUTE_LOST',
            'UF_CRM_DISPUTE_STATUS' => 'Lost'
        ]);

        \Bitrix\Crm\Timeline\CommentController::getInstance()->onCreate([
            'ENTITY_TYPE_ID' => \CCrmOwnerType::Deal,
            'ENTITY_ID' => $disputeDeal['ID'],
            'COMMENT' => "❌ DISPUTE LOST\n"
                       . "Amount deducted: $" . number_format($dispute->amount / 100, 2) . "\n"
                       . "$15 dispute fee charged"
        ]);

        // Update original Deal payment status
        $originalDealId = $disputeDeal['UF_CRM_ORIGINAL_DEAL_ID'];
        \CCrmDeal::Update($originalDealId, [
            'UF_CRM_PAYMENT_STATUS' => 'Disputed - Lost'
        ]);

        \Bitrix\Crm\Timeline\CommentController::getInstance()->onCreate([
            'ENTITY_TYPE_ID' => \CCrmOwnerType::Deal,
            'ENTITY_ID' => $originalDealId,
            'COMMENT' => "❌ Chargeback dispute LOST - payment reversed"
        ]);
    }
}

function findDisputeDealByStripeId($disputeId) {
    $result = \CCrmDeal::GetList(
        [],
        [
            'UF_CRM_DISPUTE_ID' => $disputeId

        false,

        ['ID', 'UF_CRM_ORIGINAL_DEAL_ID']
```
