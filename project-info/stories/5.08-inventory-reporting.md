# TR-27: Inventory - Reporting & Analytics

**Epic:** Inventory Management System
**Component:** Reports, Analytics & Business Intelligence
**Version:** 1.0
**Last Updated:** October 31, 2025

---

## 1. Overview

This story covers all inventory reporting, analytics, and business intelligence features. It provides visibility into inventory performance, identifies trends, detects issues, and supports data-driven decision making.

**Key Objectives:**
- Real-time inventory visibility
- Usage pattern analysis
- Cost tracking and optimization
- Loss prevention and shrinkage detection
- Performance metrics per technician
- Supplier performance tracking
- Predictive analytics for restocking
- Exportable reports
- Dashboard for management

**Target Users:**
- Owner (strategic decisions)
- Manager (operational oversight)
- Logistics (inventory management)
- Accounting (financial reporting)
- Dispatchers (resource allocation)

### 8.1. Database Views for Reporting

```sql
-- Current stock levels with value
CREATE VIEW v_inventory_current_stock AS
SELECT
  p.ID as product_id,
  p.NAME as product_name,
  p.UF_SKU as sku,
  c.NAME as category,
  s.ID as store_id,
  s.NAME as store_name,
  sp.AMOUNT as on_hand,
  p.UF_COST_TECH as unit_cost,
  (sp.AMOUNT * p.UF_COST_TECH) as total_value,
  p.UF_PRICE_CLIENT as unit_price,
  (sp.AMOUNT * p.UF_PRICE_CLIENT) as total_retail_value,
  CASE
    WHEN sp.AMOUNT = 0 THEN 'out_of_stock'
    WHEN sp.AMOUNT < p.UF_REORDER_POINT THEN 'low_stock'
    WHEN sp.AMOUNT > p.UF_MAX_STOCK THEN 'overstock'
    ELSE 'healthy'
  END as stock_status
FROM b_iblock_element p
JOIN b_catalog_store_product sp ON sp.PRODUCT_ID = p.ID
JOIN b_catalog_store s ON s.ID = sp.STORE_ID
LEFT JOIN b_iblock_section c ON c.ID = p.IBLOCK_SECTION_ID
WHERE p.IBLOCK_ID = 12 -- Products catalog
  AND p.ACTIVE = 'Y';

-- Usage by technician with variance
CREATE VIEW v_inventory_usage_by_tech AS
SELECT
  s.ID as store_id,
  s.NAME as tech_name,
  u.ID as user_id,
  COUNT(DISTINCT cpr.OWNER_ID) as jobs_count,
  SUM(cpr.QUANTITY) as total_units_used,
  SUM(cpr.QUANTITY * cpr.UF_COST_TECH) as total_cost,
  AVG(deal_totals.parts_cost) as avg_parts_per_job,
  (system_stock.expected - actual_stock.on_hand) as variance_units,
  (system_stock.expected - actual_stock.on_hand) * avg_cost as variance_value
FROM b_catalog_store s
JOIN b_user u ON u.ID = s.UF_TECHNICIAN_ID
LEFT JOIN b_crm_product_row cpr ON cpr.UF_SOURCE_STORE_ID = s.ID
  AND cpr.OWNER_TYPE = 'D'
  AND cpr.DATE_CREATE >= DATE_SUB(NOW(), INTERVAL 30 DAY)
-- ... more joins for variance calculation
GROUP BY s.ID;

-- Dead stock analysis
CREATE VIEW v_inventory_dead_stock AS
SELECT
  p.ID as product_id,
  p.NAME as product_name,
  p.UF_SKU as sku,
  s.NAME as location,
  sp.AMOUNT as quantity,
  (sp.AMOUNT * p.UF_COST_TECH) as dead_stock_value,
  DATEDIFF(NOW(), last_move.movement_date) as days_no_movement,
  last_move.movement_date as last_movement_date
FROM b_iblock_element p
JOIN b_catalog_store_product sp ON sp.PRODUCT_ID = p.ID
JOIN b_catalog_store s ON s.ID = sp.STORE_ID
LEFT JOIN (
  SELECT PRODUCT_ID, MAX(DATE_CREATE) as movement_date
  FROM b_inventory_movement
  GROUP BY PRODUCT_ID
) last_move ON last_move.PRODUCT_ID = p.ID
WHERE DATEDIFF(NOW(), last_move.movement_date) > 180
  AND sp.AMOUNT > 0
ORDER BY dead_stock_value DESC;
```

```php
namespace Local\Inventory;

class ReportingManager {

    /**
     * Get dashboard metrics
     * GET /api/inventory/reports/dashboard
     */
    public static function getDashboardMetrics($filters = []) {
        global $DB;

        $dateFrom = $filters['date_from'] ?? date('Y-m-d', strtotime('-30 days'));
        $dateTo = $filters['date_to'] ?? date('Y-m-d');

        // Total inventory value
        $valueQuery = $DB->Query("
            SELECT SUM(sp.AMOUNT * p.UF_COST_TECH) as total_value
            FROM b_catalog_store_product sp
            JOIN b_iblock_element p ON p.ID = sp.PRODUCT_ID
        ");
        $totalValue = $valueQuery->Fetch()['total_value'];

        // Stock status breakdown
        $statusQuery = $DB->Query("
            SELECT stock_status, COUNT(*) as count
            FROM v_inventory_current_stock
            GROUP BY stock_status
        ");
        $stockStatus = [];
        while ($row = $statusQuery->Fetch()) {
            $stockStatus[$row['stock_status']] = $row['count'];
        }

        // Usage trend
        $usageQuery = $DB->Query("
            SELECT DATE(DATE_CREATE) as date, SUM(QUANTITY) as units
            FROM b_inventory_movement
            WHERE MOVEMENT_TYPE = 'deal_usage'
              AND DATE_CREATE BETWEEN '{$dateFrom}' AND '{$dateTo}'
            GROUP BY DATE(DATE_CREATE)
            ORDER BY date
        ");
        $usageTrend = [];
        while ($row = $usageQuery->Fetch()) {
            $usageTrend[] = $row;
        }

        // Top movers
        $topMoversQuery = $DB->Query("
            SELECT p.UF_SKU as sku, p.NAME as name,
                   SUM(m.QUANTITY) as total_used,
                   SUM(m.QUANTITY * m.UNIT_COST) as total_value
            FROM b_inventory_movement m
            JOIN b_iblock_element p ON p.ID = m.PRODUCT_ID
            WHERE m.MOVEMENT_TYPE = 'deal_usage'
              AND m.DATE_CREATE BETWEEN '{$dateFrom}' AND '{$dateTo}'
            GROUP BY m.PRODUCT_ID
            ORDER BY total_used DESC
            LIMIT 10
        ");
        $topMovers = [];
        while ($row = $topMoversQuery->Fetch()) {
            $topMovers[] = $row;
        }

        return [
            'total_value' => $totalValue,
            'stock_status' => $stockStatus,
            'usage_trend' => $usageTrend,
            'top_movers' => $topMovers
        ];
    }

    public static function getStockLevelsReport($filters = []) {
        global $DB;

        $location = $filters['location'] ?? null;
        $category = $filters['category'] ?? null;
        $status = $filters['status'] ?? null;

        $where = ["1=1"];
        if ($location) $where[] = "store_id = {$location}";
        if ($category) $where[] = "category = '{$DB->ForSql($category)}'";
        if ($status) $where[] = "stock_status = '{$status}'";

        $query = $DB->Query("
            SELECT * FROM v_inventory_current_stock
            WHERE " . implode(" AND ", $where) . "
            ORDER BY product_name
        ");

        $report = [];
        while ($row = $query->Fetch()) {
            $report[] = $row;
        }

        return [
            'data' => $report,
            'generated_at' => date('Y-m-d H:i:s'),
            'filters' => $filters
        ];
    }

    /**
     * Get usage by technician report
     * GET /api/inventory/reports/usage-by-tech
     */
    public static function getUsageByTechReport($filters = []) {
        global $DB;

        $dateFrom = $filters['date_from'] ?? date('Y-m-d', strtotime('-30 days'));
        $dateTo = $filters['date_to'] ?? date('Y-m-d');

        $query = $DB->Query("
            SELECT
              s.ID as tech_id,
              s.NAME as tech_name,
              COUNT(DISTINCT cpr.OWNER_ID) as jobs_count,
              SUM(cpr.QUANTITY) as total_units,
              SUM(cpr.QUANTITY * cpr.UF_COST_TECH) as total_value,
              AVG(cpr.QUANTITY * cpr.UF_COST_TECH) as avg_per_job
            FROM b_catalog_store s
            LEFT JOIN b_crm_product_row cpr ON cpr.UF_SOURCE_STORE_ID = s.ID
              AND cpr.OWNER_TYPE = 'D'
              AND cpr.DATE_CREATE BETWEEN '{$dateFrom}' AND '{$dateTo}'
            WHERE s.UF_STORE_TYPE = 'container'
            GROUP BY s.ID
            ORDER BY total_value DESC
        ");

        $report = [];
        while ($row = $query->Fetch()) {
            // Calculate variance
            $variance = self::calculateTechVariance($row['tech_id'], $dateFrom, $dateTo);
            $row['variance'] = $variance;

            $report[] = $row;
        }

        return [
            'data' => $report,
            'period' => ['from' => $dateFrom, 'to' => $dateTo]
        ];
    }

    /**
     * Get dead stock report
     * GET /api/inventory/reports/dead-stock
     */
    public static function getDeadStockReport($daysThreshold = 180) {
        global $DB;

        $query = $DB->Query("
            SELECT * FROM v_inventory_dead_stock

        }

        return [
            'items' => $report,
            'total_count' => count($report),
            'total_value' => $totalValue,
            'threshold_days' => $daysThreshold
        ];
    }
```
