**Implementation - User Management API:**

```php
<?php
namespace Bitrix\Im\Admin;

use Bitrix\Main\UserTable;
use Bitrix\Im\User as ImUser;

class UserManagement
{
    /**
     * Get all users with messenger statistics
     */
    public static function getUsers($filter = [])
    {
        $users = UserTable::getList([
            'filter' => array_merge(['ACTIVE' => 'Y'], $filter),
            'select' => ['ID', 'EMAIL', 'NAME', 'LAST_NAME', 'LAST_ACTIVITY_DATE'],
            'order' => ['LAST_NAME' => 'ASC']
        ]);

        $result = [];
        while ($user = $users->fetch()) {
            $userId = $user['ID'];

            // Get messenger statistics
            $stats = self::getUserMessengerStats($userId);

            $result[] = [
                'ID' => $userId,
                'EMAIL' => $user['EMAIL'],
                'NAME' => $user['NAME'] . ' ' . $user['LAST_NAME'],
                'LAST_ACTIVE' => $user['LAST_ACTIVITY_DATE'],
                'STATS' => $stats,
                'ROLE' => self::getUserRole($userId),
                'MESSENGER_ENABLED' => ImUser::getInstance($userId)->isActive()
            ];
        }

        return $result;
    }

    /**
     * Get messenger statistics for user
     */
    private static function getUserMessengerStats($userId)
    {
        // Count chats user is member of
        $chatCount = \CIMChat::GetUserChatCount($userId);

        // Count messages sent by user
        $messageCount = \CIMMessage::GetList([
            'AUTHOR_ID' => $userId
        ], ['nTopCount' => false]);

        // Count calls participated
        $callCount = \Bitrix\Im\Call\Call::getList([
            'filter' => ['INITIATOR_ID' => $userId],
            'count_total' => true
        ]);

        return [
            'CHATS' => $chatCount,
            'MESSAGES' => $messageCount,
            'CALLS' => $callCount
        ];
    }

    /**
     * Create new user account
     */
    public static function createUser($data)
    {
        $user = new \CUser;

        $fields = [
            'EMAIL' => $data['email'],
            'LOGIN' => $data['email'],
            'NAME' => $data['first_name'],
            'LAST_NAME' => $data['last_name'],
            'PASSWORD' => \Bitrix\Main\Security\Random::getString(12),
            'CONFIRM_PASSWORD' => \Bitrix\Main\Security\Random::getString(12),
            'ACTIVE' => 'Y',
            'GROUP_ID' => [self::getGroupIdByRole($data['role'])]
        ];

        $userId = $user->Add($fields);

        if ($userId) {
            // Enable messenger for new user
            ImUser::getInstance($userId)->enable();

            // Send invitation email
            self::sendInvitationEmail($userId, $fields['PASSWORD']);

            return [
                'success' => true,
                'user_id' => $userId
            ];
        } else {
            return [
                'success' => false,
                'error' => $user->LAST_ERROR
            ];
        }
    }

    /**
     * Deactivate user messenger access
     */
    public static function deactivateUser($userId, $reason = '')
    {
        // Deactivate messenger
        ImUser::getInstance($userId)->disable();

        // Log deactivation
        self::logUserAction([
            'ACTION' => 'deactivate',
            'USER_ID' => $userId,
            'REASON' => $reason,
            'TIMESTAMP' => new \DateTime()
        ]);

        return true;
    }

    /**
     * Reactivate user messenger access
     */
    public static function reactivateUser($userId)
    {
        // Reactivate messenger
        ImUser::getInstance($userId)->enable();

        // Log reactivation
        self::logUserAction([
            'ACTION' => 'reactivate',
            'USER_ID' => $userId,
            'TIMESTAMP' => new \DateTime()
        ]);

        return true;
    }

    /**
     * Bulk deactivate users
     */
    public static function bulkDeactivate($userIds, $reason = '')
    {
        $results = [];

        foreach ($userIds as $userId) {
            $results[$userId] = self::deactivateUser($userId, $reason);
        }

        return $results;
    }

    /**
     * Get user role
     */
    private static function getUserRole($userId)
    {
        $userGroups = \CUser::GetUserGroup($userId);

        $roleMap = [
            'DISPATCHER' => 'Dispatcher',
            'TECHNICIAN' => 'Technician',
            'SHIFT_MANAGER' => 'Manager',
            'ADMINISTRATOR' => 'Administrator'
        ];

        foreach ($roleMap as $groupName => $roleName) {
            $groupId = self::getGroupIdByName($groupName);
            if (in_array($groupId, $userGroups)) {
                return $roleName;
            }
        }

        return 'Employee';
    }

    private static function getGroupIdByName($name)
    {
        $group = \CGroup::GetList(
            $by = 'NAME',
            $order = 'ASC',
            ['NAME' => $name]
        )->Fetch();

        return $group ? $group['ID'] : null;
    }

    private static function getGroupIdByRole($roleName)
    {
        $roleMap = [
            'dispatcher' => 'DISPATCHER',
            'technician' => 'TECHNICIAN',
            'manager' => 'SHIFT_MANAGER',
            'admin' => 'ADMINISTRATOR'
        ];

        $groupName = $roleMap[$roleName] ?? 'EMPLOYEE';
        return self::getGroupIdByName($groupName);
    }

    private static function sendInvitationEmail($userId, $password)
    {
        $user = \CUser::GetByID($userId)->Fetch();

        $message = "Welcome to Bitrix24!\n\n";
        $message .= "Your account has been created:\n";
        $message .= "Email: {$user['EMAIL']}\n";
        $message .= "Temporary Password: {$password}\n\n";
        $message .= "Please log in and change your password.\n";

        \CEvent::Send('USER_INVITATION', SITE_ID, [
            'EMAIL_TO' => $user['EMAIL'],
            'MESSAGE' => $message
        ]);
    }

    private static function logUserAction($data)
    {
        global $DB;

        $DB->Add('b_im_user_action_log', $data);
    }
}
```

**Implementation - Group Approval Workflow:**

```php
<?php
namespace Bitrix\Im\Admin;

use Bitrix\Im\V2\Chat;

class GroupApprovalWorkflow
{
    /**
     * Request group creation (if approval required)
     */
    public static function requestGroupCreation($data, $requesterId)
    {
        // Check if approval is required
        $approvalRequired = \COption::GetOptionString('im', 'group_approval_required', 'N');

        // Check if requester is auto-approved
        $autoApproved = self::isUserAutoApproved($requesterId);

        if ($approvalRequired === 'Y' && !$autoApproved) {
            // Create approval request
            $requestId = self::createApprovalRequest($data, $requesterId);

            // Notify administrators
            self::notifyAdministrators($requestId);

            return [
                'success' => true,
                'status' => 'pending_approval',
                'request_id' => $requestId,
                'message' => 'Group creation request submitted for approval'
            ];
        } else {
            // Create group immediately
            $chatId = self::createGroup($data, $requesterId);

            return [
                'success' => true,
                'status' => 'created',
                'chat_id' => $chatId
            ];
        }
    }

    /**
     * Create approval request
     */
    private static function createApprovalRequest($data, $requesterId)
    {
        global $DB;

        $requestId = $DB->Add('b_im_group_approval_request', [
            'NAME' => $data['name'],
            'DESCRIPTION' => $data['description'],
            'TYPE' => $data['type'],
            'MEMBERS' => serialize($data['members']),
            'REQUESTER_ID' => $requesterId,
            'STATUS' => 'pending',
            'DATE_CREATED' => new \DateTime()
        ]);

        return $requestId;
    }

    /**
     * Approve group creation request
     */
    public static function approveRequest($requestId, $adminId)
    {
        global $DB;

        // Get request details
        $request = $DB->Query("SELECT * FROM b_im_group_approval_request WHERE ID = {$requestId}")->Fetch();

        if (!$request || $request['STATUS'] !== 'pending') {
            return ['success' => false, 'error' => 'Request not found or already processed'];
        }

        // Create group
        $data = [
            'name' => $request['NAME'],
            'description' => $request['DESCRIPTION'],
            'type' => $request['TYPE'],
            'members' => unserialize($request['MEMBERS'])
        ];

        $chatId = self::createGroup($data, $request['REQUESTER_ID']);

        // Update request status
        $DB->Update('b_im_group_approval_request', [
            'STATUS' => 'approved',
            'CHAT_ID' => $chatId,
            'APPROVED_BY' => $adminId,
            'DATE_PROCESSED' => new \DateTime()
        ], "WHERE ID = {$requestId}");

        // Notify requester
        self::notifyRequester($request['REQUESTER_ID'], $request['NAME'], 'approved');

        return [
            'success' => true,
            'chat_id' => $chatId
        ];
    }

    /**
     * Reject group creation request
     */
    public static function rejectRequest($requestId, $adminId, $reason = '')
    {
        global $DB;

        // Get request details
        $request = $DB->Query("SELECT * FROM b_im_group_approval_request WHERE ID = {$requestId}")->Fetch();

        if (!$request || $request['STATUS'] !== 'pending') {
            return ['success' => false, 'error' => 'Request not found or already processed'];
        }

        // Update request status
        $DB->Update('b_im_group_approval_request', [
            'STATUS' => 'rejected',
            'REJECTED_BY' => $adminId,
            'REJECTION_REASON' => $reason,
            'DATE_PROCESSED' => new \DateTime()
        ], "WHERE ID = {$requestId}");

        // Notify requester
        self::notifyRequester($request['REQUESTER_ID'], $request['NAME'], 'rejected', $reason);

        return ['success' => true];
    }

    /**
     * Create group chat
     */
    private static function createGroup($data, $creatorId)
    {
        $chatResult = Chat\ChatFactory::getInstance()->addChat([
            'TYPE' => $data['type'] ?? Chat::IM_TYPE_OPEN,
            'TITLE' => $data['name'],
            'DESCRIPTION' => $data['description'],
            'USERS' => $data['members'],
            'AUTHOR_ID' => $creatorId
        ]);

        if ($chatResult->isSuccess()) {
            return $chatResult->getResult()['CHAT_ID'];
        }

        return null;
    }

    /**
     * Check if user is auto-approved for group creation
     */
    private static function isUserAutoApproved($userId)
    {
        $autoApprovedRoles = ['SHIFT_MANAGER', 'ADMINISTRATOR'];

        $userGroups = \CUser::GetUserGroup($userId);

        foreach ($autoApprovedRoles as $roleName) {
            $groupId = self::getGroupIdByName($roleName);
            if (in_array($groupId, $userGroups)) {
                return true;
            }
        }

        return false;
    }

    /**
     * Notify administrators of new approval request
     */
    private static function notifyAdministrators($requestId)
    {
        // Get all administrators
        $admins = \CUser::GetList(
            $by = 'ID',
            $order = 'ASC',
            ['GROUPS_ID' => [self::getGroupIdByName('ADMINISTRATOR')]]
        );

        while ($admin = $admins->Fetch()) {
            // Send notification via Bitrix24 notification system
            \CIMNotify::Add([
                'TO_USER_ID' => $admin['ID'],
                'FROM_USER_ID' => 1, // System
                'NOTIFY_TYPE' => IM_NOTIFY_SYSTEM,
                'NOTIFY_MODULE' => 'im',
                'NOTIFY_EVENT' => 'group_approval_request',
                'NOTIFY_MESSAGE' => "New group creation request #{$requestId} awaiting approval"
            ]);
        }
    }

    /**
     * Notify requester of decision
     */
    private static function notifyRequester($userId, $groupName, $status, $reason = '')
    {
        $message = $status === 'approved'
            ? "Your group creation request '{$groupName}' has been approved!"
            : "Your group creation request '{$groupName}' has been rejected. Reason: {$reason}";

        \CIMNotify::Add([
            'TO_USER_ID' => $userId,
            'FROM_USER_ID' => 1, // System
            'NOTIFY_TYPE' => IM_NOTIFY_SYSTEM,
            'NOTIFY_MODULE' => 'im',
            'NOTIFY_EVENT' => 'group_approval_decision',
            'NOTIFY_MESSAGE' => $message
        ]);
    }

    private static function getGroupIdByName($name)
    {
        $group = \CGroup::GetList(
            $by = 'NAME',
            $order = 'ASC',
            ['NAME' => $name]
        )->Fetch();

        return $group ? $group['ID'] : null;
    }
}
```

**Implementation - Group Moderation:**

```php
<?php
namespace Bitrix\Im\Admin;

use Bitrix\Im\V2\Chat;

class GroupModeration
{
    /**
     * Get all groups with statistics
     */
    public static function getAllGroups($filter = [])
    {
        $chats = Chat\ChatTable::getList([
            'filter' => array_merge(['TYPE' => Chat::IM_TYPE_OPEN], $filter),
            'select' => ['ID', 'TITLE', 'AUTHOR_ID', 'DATE_CREATE', 'ENTITY_DATA_1'],
            'order' => ['DATE_CREATE' => 'DESC']
        ]);

        $result = [];
        while ($chat = $chats->fetch()) {
            $chatId = $chat['ID'];

            // Get statistics
            $stats = self::getChatStatistics($chatId);

            $result[] = [
                'ID' => $chatId,
                'NAME' => $chat['TITLE'],
                'OWNER_ID' => $chat['AUTHOR_ID'],
                'OWNER_NAME' => self::getUserName($chat['AUTHOR_ID']),
                'DATE_CREATED' => $chat['DATE_CREATE'],
                'STATS' => $stats,
                'IS_ARCHIVED' => $chat['ENTITY_DATA_1'] === 'archived'
            ];
        }

        return $result;
    }

    /**
     * Get chat statistics
     */
    private static function getChatStatistics($chatId)
    {
        // Count members
        $memberCount = Chat\RelationTable::getCount([
            'CHAT_ID' => $chatId
        ]);

        // Count messages
        $messageCount = \CIMMessage::GetList([
            'CHAT_ID' => $chatId
        ], ['nTopCount' => false]);

        // Count files
        $fileCount = \Bitrix\Disk\AttachedObject::getList([
            'filter' => ['ENTITY_ID' => $chatId, 'ENTITY_TYPE' => \Bitrix\Disk\Uf\FileUserType::className()]
        ])->getSelectedRowsCount();

        // Get last activity
        $lastMessage = \CIMMessage::GetList([
            'CHAT_ID' => $chatId
        ], ['nTopCount' => 1]);

        return [
            'MEMBERS' => $memberCount,
            'MESSAGES' => $messageCount,
            'FILES' => $fileCount,
            'LAST_ACTIVITY' => $lastMessage ? $lastMessage[0]['DATE_CREATE'] : null
        ];
    }

    /**
     * Archive group chat
     */
    public static function archiveGroup($chatId, $adminId, $reason = '')
    {
        // Mark chat as archived
        Chat\ChatTable::update($chatId, [
            'ENTITY_DATA_1' => 'archived',
            'ENTITY_DATA_2' => json_encode([
                'archived_by' => $adminId,
                'archived_date' => date('Y-m-d H:i:s'),
                'reason' => $reason
            ])
        ]);

        // Notify members
        self::notifyGroupMembers($chatId, "This group has been archived by an administrator. Reason: {$reason}");

        // Log action
        self::logModerationAction([
            'ACTION' => 'archive_group',
            'CHAT_ID' => $chatId,
            'ADMIN_ID' => $adminId,
            'REASON' => $reason,
            'TIMESTAMP' => new \DateTime()
        ]);

        return true;
    }

    /**
     * Restore archived group
     */
    public static function restoreGroup($chatId, $adminId)
    {
        Chat\ChatTable::update($chatId, [
            'ENTITY_DATA_1' => null,
            'ENTITY_DATA_2' => null
        ]);

        // Notify members
        self::notifyGroupMembers($chatId, "This group has been restored by an administrator.");

        // Log action
        self::logModerationAction([
            'ACTION' => 'restore_group',
            'CHAT_ID' => $chatId,
            'ADMIN_ID' => $adminId,
            'TIMESTAMP' => new \DateTime()
        ]);

        return true;
    }

    /**
     * Delete group permanently
     */
    public static function deleteGroup($chatId, $adminId, $reason = '')
    {
        // Get chat details for logging
        $chat = Chat\ChatTable::getById($chatId)->fetch();

        // Delete chat (this will also delete messages and relations)
        Chat\Chat::getInstance($chatId)->deleteChat();

        // Log deletion
        self::logModerationAction([
            'ACTION' => 'delete_group',
            'CHAT_ID' => $chatId,
            'CHAT_NAME' => $chat['TITLE'],
            'ADMIN_ID' => $adminId,
            'REASON' => $reason,
            'TIMESTAMP' => new \DateTime()
        ]);

        return true;
    }

    /**
     * Add member to group
     */
    public static function addMember($chatId, $userId, $adminId)
    {
        $chat = Chat\Chat::getInstance($chatId);
        $chat->addUsers([$userId]);

        // Notify user
        \CIMNotify::Add([
            'TO_USER_ID' => $userId,
            'FROM_USER_ID' => $adminId,
            'NOTIFY_TYPE' => IM_NOTIFY_SYSTEM,
            'NOTIFY_MESSAGE' => "You have been added to group: {$chat->getTitle()}"
        ]);

        return true;
    }

    /**
     * Remove member from group
     */
    public static function removeMember($chatId, $userId, $adminId, $reason = '')
    {
        $chat = Chat\Chat::getInstance($chatId);
        $chat->deleteUser($userId);

        // Notify user
        \CIMNotify::Add([
            'TO_USER_ID' => $userId,
            'FROM_USER_ID' => $adminId,
            'NOTIFY_TYPE' => IM_NOTIFY_SYSTEM,
            'NOTIFY_MESSAGE' => "You have been removed from group: {$chat->getTitle()}. Reason: {$reason}"
        ]);

        return true;
    }

    /**
     * Transfer group ownership
     */
    public static function transferOwnership($chatId, $newOwnerId, $adminId)
    {
        Chat\ChatTable::update($chatId, [
            'AUTHOR_ID' => $newOwnerId
        ]);

        $chat = Chat\Chat::getInstance($chatId);

        // Notify new owner
        \CIMNotify::Add([
            'TO_USER_ID' => $newOwnerId,
            'FROM_USER_ID' => $adminId,
            'NOTIFY_TYPE' => IM_NOTIFY_SYSTEM,
            'NOTIFY_MESSAGE' => "You are now the owner of group: {$chat->getTitle()}"
        ]);

        return true;
    }

    private static function notifyGroupMembers($chatId, $message)
    {
        $members = Chat\RelationTable::getList([
            'filter' => ['CHAT_ID' => $chatId],
            'select' => ['USER_ID']
        ]);

        while ($member = $members->fetch()) {
            \CIMNotify::Add([
                'TO_USER_ID' => $member['USER_ID'],
                'FROM_USER_ID' => 1, // System
                'NOTIFY_TYPE' => IM_NOTIFY_SYSTEM,
                'NOTIFY_MESSAGE' => $message
            ]);
        }
    }

    private static function getUserName($userId)
    {
        $user = \CUser::GetByID($userId)->Fetch();
        return $user ? $user['NAME'] . ' ' . $user['LAST_NAME'] : 'Unknown';
    }

    private static function logModerationAction($data)
    {
        global $DB;
        $DB->Add('b_im_moderation_log', $data);
    }
}
```

**Implementation - Audit Logger:**

```php
<?php
namespace Bitrix\Im\Admin;

class AuditLogger
{
    /**
     * Log administrative action
     */
    public static function log($actionType, $details = [])
    {
        global $USER, $DB;

        $logData = [
            'ACTION_TYPE' => $actionType,
            'ADMIN_USER_ID' => $USER->GetID(),
            'TARGET_ENTITY_TYPE' => $details['entity_type'] ?? null,
            'TARGET_ENTITY_ID' => $details['entity_id'] ?? null,
            'DETAILS' => json_encode($details),
            'IP_ADDRESS' => $_SERVER['REMOTE_ADDR'] ?? null,
            'TIMESTAMP' => new \DateTime()
        ];

        $DB->Add('b_im_audit_log', $logData);
    }

    /**
     * Get audit log entries
     */
    public static function getLog($filter = [], $limit = 25, $offset = 0)
    {
        global $DB;

        $where = ['1=1'];

        if (!empty($filter['date_from'])) {
            $where[] = "TIMESTAMP >= '{$filter['date_from']}'";
        }

        if (!empty($filter['date_to'])) {
            $where[] = "TIMESTAMP <= '{$filter['date_to']}'";
        }

        if (!empty($filter['action_type'])) {
            $where[] = "ACTION_TYPE = '{$filter['action_type']}'";
        }

        if (!empty($filter['admin_user_id'])) {
            $where[] = "ADMIN_USER_ID = {$filter['admin_user_id']}";
        }

        $whereClause = implode(' AND ', $where);

        $query = "
            SELECT *
            FROM b_im_audit_log
            WHERE {$whereClause}
            ORDER BY TIMESTAMP DESC
            LIMIT {$limit} OFFSET {$offset}
        ";

        $result = $DB->Query($query);

        $logs = [];
        while ($log = $result->Fetch()) {
            $logs[] = [
                'ID' => $log['ID'],
                'ACTION_TYPE' => $log['ACTION_TYPE'],
                'ADMIN_NAME' => self::getUserName($log['ADMIN_USER_ID']),
                'DETAILS' => json_decode($log['DETAILS'], true),
                'IP_ADDRESS' => $log['IP_ADDRESS'],
                'TIMESTAMP' => $log['TIMESTAMP']
            ];
        }

        return $logs;
    }

    /**
     * Export audit log to CSV
     */
    public static function exportToCSV($filter = [])
    {
        $logs = self::getLog($filter, 10000); // Get up to 10,000 records

        $csv = [];
        $csv[] = ['Timestamp', 'Admin', 'Action', 'Details', 'IP Address'];

        foreach ($logs as $log) {
            $csv[] = [
                $log['TIMESTAMP'],
                $log['ADMIN_NAME'],
                $log['ACTION_TYPE'],
                json_encode($log['DETAILS']),
                $log['IP_ADDRESS']
            ];
        }

        return $csv;
    }

    /**
     * Log call recording access
     */
    public static function logRecordingAccess($recordingId, $accessType = 'view')
    {
        global $USER, $DB;

        $DB->Add('b_im_recording_access_log', [
            'RECORDING_ID' => $recordingId,
            'ACCESSED_BY' => $USER->GetID(),
            'ACCESS_TYPE' => $accessType,
            'TIMESTAMP' => new \DateTime(),
            'IP_ADDRESS' => $_SERVER['REMOTE_ADDR'] ?? null
        ]);
    }

    private static function getUserName($userId)
    {
        if (!$userId) return 'System';

        $user = \CUser::GetByID($userId)->Fetch();
        return $user ? $user['NAME'] . ' ' . $user['LAST_NAME'] : 'Unknown';
    }
}
```

## Risks & Mitigation

| Risk | Impact | Mitigation Strategy |
|------|--------|---------------------|
| **Accidental group deletion** | High - data loss | - Require confirmation with group name<br>- Implement soft delete (archive first)<br>- Audit log all deletions<br>- 30-day recovery window |
| **Unauthorized access to recordings** | High - privacy | - Strict access controls<br>- Audit all recording access<br>- Automatic access revocation<br>- Legal hold capability |
| **Audit log storage growth** | Medium - storage | - Set retention policies<br>- Archive old logs<br>- Compress archived logs<br>- Monitor storage usage |
| **Permission configuration errors** | Medium - security | - Test all permission scenarios<br>- Document permission matrix<br>- Regular permission audits<br>- Role-based access templates |
