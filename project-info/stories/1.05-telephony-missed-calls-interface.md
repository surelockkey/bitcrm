## 3. Missed Call Detection

### 3.1 What Qualifies as "Missed Call"

```php
class MissedCallDetector {
    /**
     * Determines if call is "missed"
     *
     * Missed call criteria:
     * 1. Inbound call (direction = 'inbound')
     * 2. Status = 'no-answer', 'busy', or 'canceled'
     * 3. Duration = 0 OR answered = false
     */
    public function isMissedCall($callData) {
        // Ignore outbound calls
        if ($callData['direction'] !== 'inbound') {
            return false;
        }

        // Check call status
        $missedStatuses = ['no-answer', 'busy', 'canceled'];
        if (!in_array($callData['status'], $missedStatuses)) {
            return false;
        }

        // Check if call was answered
        if ($callData['answered'] === true && $callData['duration'] > 0) {
            return false;
        }

        return true;
    }
}
```

### 3.2 Twilio Webhook for Missed Calls

```php
// /local/webhooks/twilio/missed_call.php

use Bitrix\Main\Application;
use Bitrix\Crm\Timeline;

class TwilioMissedCallHandler {
    public function handleWebhook() {
        $request = Application::getInstance()->getContext()->getRequest();

        $callSid = $request->get('CallSid');
        $from = $request->get('From');
        $to = $request->get('To');
        $callStatus = $request->get('CallStatus');
        $duration = (int)$request->get('CallDuration');
        $answered = $request->get('AnsweredBy');

        // Check if this is a missed call
        $detector = new MissedCallDetector();
        $isMissed = $detector->isMissedCall([
            'direction' => 'inbound',
            'status' => $callStatus,
            'answered' => ($answered === 'human' || $answered === 'dispatcher'),
            'duration' => $duration
        ]);

        if (!$isMissed) {
            return; // Not a missed call, ignore
        }

        // Check if contact exists
        $contact = $this->findOrCreateContact($from);

        // Create missed call record
        global $DB;
        $DB->Query("
            INSERT INTO b_crm_missed_calls (
                TWILIO_SID,
                FROM_NUMBER,
                TO_NUMBER,
                CONTACT_ID,
                DEAL_ID,
                STATUS,
                DATE_CREATED,
                DATE_CALLBACK
            ) VALUES (
                '{$DB->ForSql($callSid)}',
                '{$DB->ForSql($from)}',
                '{$DB->ForSql($to)}',
                {$contact['ID']},
                NULL,
                'PENDING',
                NOW(),
                NULL
            )
        ");

        $missedCallId = $DB->LastID();

        // Add to CRM timeline
        Timeline\ActivityController::getInstance()->addMissedCall([
            'AUTHOR_ID' => 0, // System
            'SETTINGS' => [
                'CALLER' => $from,
                'CALLED_NUMBER' => $to,
                'TWILIO_SID' => $callSid,
                'MISSED_CALL_ID' => $missedCallId
            ],
            'BINDINGS' => [
                [
                    'ENTITY_TYPE_ID' => \CCrmOwnerType::Contact,
                    'ENTITY_ID' => $contact['ID']
                ]
            ]
        ]);

        // Send real-time notification to dispatchers
        $this->notifyDispatchers([
            'missedCallId' => $missedCallId,
            'from' => $from,
            'contactName' => $contact['NAME'] . ' ' . $contact['LAST_NAME']
        ]);

        // Auto-create deal draft if new contact
        if (!$contact['EXISTING']) {
            $this->createDealDraft($contact['ID'], $missedCallId);
        }
    }

    private function notifyDispatchers($data) {
        // Use Bitrix24 Pull & Push for real-time notifications
        \Bitrix\Pull\Event::add(
            $this->getDispatcherUserIds(),
            [
                'module_id' => 'crm',
                'command' => 'missed_call_new',
                'params' => $data
            ]
        );
    }
}
```

### 4.2 Database Schema

```sql
CREATE TABLE b_crm_missed_calls (
    ID INT PRIMARY KEY AUTO_INCREMENT,

    -- Twilio data
    TWILIO_SID VARCHAR(50) UNIQUE NOT NULL,
    FROM_NUMBER VARCHAR(20) NOT NULL,
    TO_NUMBER VARCHAR(20) NOT NULL,

    -- CRM links
    CONTACT_ID INT,
    DEAL_ID INT,

    -- Status tracking
    STATUS VARCHAR(20) DEFAULT 'PENDING',  -- 'PENDING', 'CALLED_BACK', 'RESOLVED', 'IGNORED'

    -- Callback tracking
    DATE_CREATED DATETIME NOT NULL,
    DATE_CALLBACK DATETIME,
    CALLBACK_USER_ID INT,
    CALLBACK_CALL_SID VARCHAR(50),

    -- Notes
    NOTES TEXT,

    INDEX idx_status (STATUS, DATE_CREATED),
    INDEX idx_contact (CONTACT_ID),
    INDEX idx_deal (DEAL_ID),
    INDEX idx_date (DATE_CREATED),
    INDEX idx_twilio (TWILIO_SID),

    FOREIGN KEY (CONTACT_ID) REFERENCES b_crm_contact(ID) ON DELETE SET NULL,
    FOREIGN KEY (DEAL_ID) REFERENCES b_crm_deal(ID) ON DELETE SET NULL
);

-- Index for dashboard queries (critical for performance)
CREATE INDEX idx_dashboard ON b_crm_missed_calls (STATUS, DATE_CREATED DESC);
```

### 4.3 Frontend Component

```javascript
// React component for missed calls dashboard
class MissedCallsDashboard extends React.Component {
    constructor(props) {
        super(props);

        this.state = {
            missedCalls: [],
            filter: 'PENDING',
            sortBy: 'DATE_DESC',
            selectedCalls: [],
            autoRefresh: true
        };
    }

    componentDidMount() {
        this.loadMissedCalls();

        // Subscribe to real-time updates
        BX.PULL.subscribe({
            moduleId: 'crm',
            command: 'missed_call_new',
            callback: (data) => {
                this.addNewMissedCall(data);
                this.showNotification(data);
            }
        });

        // Auto-refresh every 30 seconds
        if (this.state.autoRefresh) {
            this.refreshInterval = setInterval(() => {
                this.loadMissedCalls();
            }, 30000);
        }
    }

    componentWillUnmount() {
        if (this.refreshInterval) {
            clearInterval(this.refreshInterval);
        }
    }

    loadMissedCalls() {
        BX.ajax.runAction('crm:missedcalls.list', {
            data: {
                filter: this.state.filter,
                sortBy: this.state.sortBy
            }
        }).then((response) => {
            this.setState({ missedCalls: response.data.items });
        });
    }

    handleCallBack(missedCallId, contactPhone, contactId) {
        // Initiate callback through Twilio
        BX.ajax.runAction('crm:missedcalls.callback', {
            data: {
                missedCallId: missedCallId,
                contactPhone: contactPhone,
                contactId: contactId
            }
        }).then((response) => {
            if (response.data.success) {
                // Update UI to show "Calling..."
                this.updateCallStatus(missedCallId, 'CALLING');

                // Open deal/contact card in sidebar
                BX.SidePanel.Instance.open(`/crm/contact/details/${contactId}/`);
            }
        });
    }

    handleBulkAction(action) {
        const selectedIds = this.state.selectedCalls;

        BX.ajax.runAction('crm:missedcalls.bulkAction', {
            data: {
                action: action, // 'MARK_RESOLVED', 'ASSIGN', 'EXPORT'
                ids: selectedIds
            }
        }).then(() => {
            this.loadMissedCalls();
            this.setState({ selectedCalls: [] });
        });
    }

    showNotification(data) {
        // Browser notification (if permission granted)
        if (Notification.permission === 'granted') {
            new Notification('Missed Call', {
                body: `${data.contactName} - ${data.from}`,
                icon: '/bitrix/images/crm/missed_call_icon.png',
                tag: 'missed-call-' + data.missedCallId
            });
        }

        // Sound notification
        const audio = new Audio('/bitrix/sounds/missed_call.mp3');
        audio.play();
    }

    render() {
        const { missedCalls, filter } = this.state;

        return (
            <div className="missed-calls-dashboard">
                <div className="header">
                    <h2>Missed Calls Dashboard</h2>
                    <div className="auto-refresh">
                        <input
                            type="checkbox"
                            checked={this.state.autoRefresh}
                            onChange={(e) => this.setState({ autoRefresh: e.target.checked })}
                        />
                        Auto-refresh
                    </div>
                </div>

                <div className="filters">
                    <select value={filter} onChange={(e) => this.handleFilterChange(e.target.value)}>
                        <option value="ALL">All</option>
                        <option value="PENDING">Pending</option>
                        <option value="CALLED_BACK">Called Back</option>
                        <option value="RESOLVED">Resolved</option>
                    </select>
                </div>

                <div className="missed-calls-list">
                    {missedCalls.map(call => (
                        <MissedCallCard
                            key={call.ID}
                            call={call}
                            onCallBack={() => this.handleCallBack(call.ID, call.FROM_NUMBER, call.CONTACT_ID)}
                            onSendSMS={() => this.handleSendSMS(call.CONTACT_ID, call.FROM_NUMBER)}
                            onResolve={() => this.handleResolve(call.ID)}
                        />
                    ))}
                </div>
            </div>
        );
    }
}
```

## 5. One-Click Callback

### 5.1 Callback Flow

```
User clicks "📞 Call Back"
         ↓
1. Backend initiates Twilio call to dispatcher
         ↓
2. Dispatcher picks up
         ↓
3. Twilio automatically calls client
         ↓
4. Connection established
         ↓
5. Missed call status → "CALLED_BACK"
         ↓
6. New call record created in timeline
```

### 5.2 Backend Implementation

```php
class MissedCallCallbackController {
    /**
     * Initiates callback for missed call
     */
    public function callbackAction($missedCallId, $contactPhone, $contactId) {
        global $USER, $DB;

        // Get current dispatcher's phone
        $dispatcherPhone = $this->getDispatcherPhone($USER->GetID());

        if (!$dispatcherPhone) {
            return [
                'success' => false,
                'error' => 'Dispatcher phone not configured'
            ];
        }

        // Get service area number for callback
        $serviceAreaNumber = $this->getServiceAreaNumber($contactPhone);

        // Create Twilio call
        $twilioClient = new Twilio\Rest\Client(TWILIO_SID, TWILIO_TOKEN);

        try {
            // Step 1: Call dispatcher first
            $call = $twilioClient->calls->create(
                $dispatcherPhone, // To
                $serviceAreaNumber, // From (service area number)
                [
                    'url' => SITE_URL . '/local/webhooks/twilio/callback_connect.php?contactPhone=' . urlencode($contactPhone) . '&missedCallId=' . $missedCallId,
                    'statusCallback' => SITE_URL . '/local/webhooks/twilio/callback_status.php',
                    'statusCallbackEvent' => ['initiated', 'ringing', 'answered', 'completed'],
                    'record' => 'record-from-answer',
                    'recordingStatusCallback' => SITE_URL . '/local/webhooks/twilio/recording_complete.php'
                ]
            );

            // Update missed call record
            $DB->Query("
                UPDATE b_crm_missed_calls
                SET
                    STATUS = 'CALLING',
                    CALLBACK_USER_ID = {$USER->GetID()},
                    DATE_CALLBACK = NOW(),
                    CALLBACK_CALL_SID = '{$DB->ForSql($call->sid)}'
                WHERE ID = {$missedCallId}
            ");

            return [
                'success' => true,
                'callSid' => $call->sid,
                'status' => $call->status
            ];

        } catch (\Exception $e) {
            return [
                'success' => false,
                'error' => $e->getMessage()
            ];
        }
    }
}
```

## 6. Auto-Draft Creation

### 6.1 When to Create Auto-Draft

```php
class AutoDraftCreator {
    /**
     * Creates draft deal for new missed call
     */
    public function createDraftForMissedCall($contactId, $missedCallId, $fromNumber) {
        // Check if draft already exists for this contact
        $existingDraft = $this->findOpenDraft($contactId);
        if ($existingDraft) {
            return $existingDraft['ID']; // Don't create duplicate
        }

        // Determine service area based on called number
        $serviceArea = $this->getServiceAreaFromNumber($fromNumber);

        // Create deal draft
        $deal = new \CCrmDeal(false);

        $fields = [
            'TITLE' => 'Missed Call - ' . date('m/d/Y H:i'),
            'CONTACT_ID' => $contactId,
            'STAGE_ID' => 'NEW',
            'OPENED' => 'Y',
            'IS_MANUAL_OPPORTUNITY' => 'N',
            'CURRENCY_ID' => 'USD',

            // Custom fields
            'UF_CRM_SOURCE' => 'MISSED_CALL',
            'UF_CRM_SERVICE_AREA' => $serviceArea,
            'UF_CRM_MISSED_CALL_ID' => $missedCallId,
            'UF_CRM_IS_DRAFT' => 'Y',

            // Assigned to (round-robin or lead routing logic)
            'ASSIGNED_BY_ID' => $this->getNextAvailableDispatcher()
        ];

        $dealId = $deal->Add($fields);

        if ($dealId) {
            // Link missed call to deal
            global $DB;
            $DB->Query("
                UPDATE b_crm_missed_calls
                SET DEAL_ID = {$dealId}
                WHERE ID = {$missedCallId}
            ");

            // Add timeline entry
            \Bitrix\Crm\Timeline\ActivityController::getInstance()->addNote([
                'OWNER_TYPE_ID' => \CCrmOwnerType::Deal,
                'OWNER_ID' => $dealId,
                'AUTHOR_ID' => 0, // System
                'TEXT' => 'Auto-draft created from missed call',
                'SETTINGS' => [
                    'MISSED_CALL_ID' => $missedCallId
                ]
            ]);
        }

        return $dealId;
    }

    /**
     * Round-robin or lead routing
     */
    private function getNextAvailableDispatcher() {
        // Simple round-robin based on last assignment
        global $DB;

        $result = $DB->Query("
            SELECT USER_ID, COUNT(*) as deal_count
            FROM b_crm_deal
            WHERE STAGE_ID = 'NEW'
              AND DATE_CREATE > DATE_SUB(NOW(), INTERVAL 1 DAY)
              AND USER_ID IN (SELECT ID FROM b_user WHERE UF_DEPARTMENT = 'DISPATCH')
            GROUP BY USER_ID
            ORDER BY deal_count ASC
            LIMIT 1
        ")->Fetch();

        return $result ? $result['USER_ID'] : 1; // Default to admin if no dispatchers
    }
}
```

### 7.1 Dashboard Query Optimization

```sql
-- Main dashboard query (must be < 50ms)
EXPLAIN SELECT
    mc.ID,
    mc.TWILIO_SID,
    mc.FROM_NUMBER,
    mc.TO_NUMBER,
    mc.STATUS,
    mc.DATE_CREATED,
    CONCAT(c.NAME, ' ', c.LAST_NAME) as CONTACT_NAME,
```

### 7.2 Caching Strategy

```php
class MissedCallsCache {
    private $cacheTime = 30; // 30 seconds

    /**
     * Get missed calls with cache
     */
    public function getList($filter) {
        $cacheKey = 'missed_calls_' . md5($filter);

        $cache = \Bitrix\Main\Data\Cache::createInstance();

        if ($cache->initCache($this->cacheTime, $cacheKey)) {
            return $cache->getVars();
        }

        $cache->startDataCache();

        // Query database
        $result = $this->queryDatabase($filter);

        $cache->endDataCache($result);

        return $result;
    }

    /**
     * Invalidate cache on new missed call
     */
    public function invalidate() {
        $cache = \Bitrix\Main\Data\Cache::createInstance();
        $cache->cleanDir('/crm/missed_calls/');
    }
}
```

```javascript
// Use Bitrix24 Pull & Push для real-time updates
class MissedCallsRealtime {
    constructor() {
        this.subscribe();
    }

    subscribe() {
        BX.PULL.subscribe({
            moduleId: 'crm',
            command: 'missed_call_new',
            callback: (data) => {
                this.handleNewMissedCall(data);
            }
        });

        BX.PULL.subscribe({
            moduleId: 'crm',
            command: 'missed_call_status_changed',
            callback: (data) => {
                this.handleStatusChange(data);
            }
        });
    }

    handleNewMissedCall(data) {
        // Add to top of list without full refresh
        const newCard = this.createMissedCallCard(data);
        document.querySelector('.missed-calls-list').prepend(newCard);

        // Show notification
        this.showNotification(data);

        // Play sound
        this.playSound();
    }

    handleStatusChange(data) {
        // Update specific card
        const card = document.querySelector(`[data-missed-call-id="${data.missedCallId}"]`);
        if (card) {
            card.setAttribute('data-status', data.newStatus);
            card.querySelector('.status').textContent = data.newStatus;
        }
    }
}
```
